            function shuffle(arr) {
                const a = [...arr];
                for(let i=a.length-1;i>0;i--) { const j=Math.floor(Math.random()*(i+1)); [a[i],a[j]]=[a[j],a[i]]; }

                }
                if(t==='duizi') {
                    if(h1.cv.pr > h2.cv.pr) return {winner:p1,reason:'对子更大'};
                    if(h2.cv.pr > h1.cv.pr) return {winner:p2,reason:'对子更大'};
                    if(h1.cv.k > h2.cv.k) return {winner:p1,reason:'对子相同，单牌更大'};
                    if(h2.cv.k > h1.cv.k) return {winner:p2,reason:'对子相同，单牌更大'};
                    return {winner:null,reason:'平局'};
                }
                return {winner:null,reason:'平局'};
            }
            function getActivePlayers() { return players.filter(p=>p.status==='active'); }
            function getActiveAIs() { return players.filter((p,i)=>i!==humanPlayerIndex && p.status==='active'); }

            function renderAll() {
                const ais = players.filter((_,i)=>i!==humanPlayerIndex);
                aiRow.innerHTML = '';
                ais.forEach((p,idx)=>{
                    const realIdx = players.indexOf(p);
                    const slot = document.createElement('div');
                    slot.className = 'player-slot';
                    if(p.status==='folded') slot.classList.add('folded');
                    if(p.status==='eliminated') slot.classList.add('eliminated');
                    slot.id = 'aiSlot'+realIdx;
                    slot.innerHTML = `
                        <div class="player-avatar">${p.avatar}</div>
                        <div class="player-name">${p.name}</div>
                        <div class="player-chips">💰 ${p.chips}</div>
                        <div class="player-status ${p.status==='folded'?'fold-status':''}">${p.status==='folded'?'已弃牌':p.status==='eliminated'?'已出局':''}</div>
                        <div class="mini-cards" id="miniCards${realIdx}"></div>
                    `;
                    aiRow.appendChild(slot);
                    const mc = document.getElementById('miniCards'+realIdx);
                    if(p.cards && p.cards.length===3) {
                        const reveal = (p.status==='eliminated' || p.status==='folded');
                        p.cards.forEach(c=>{
                            const el=document.createElement('div');
                            el.className='mini-card';
                            if(reveal || !gameInProgress) {
                                el.classList.add('face-up');
                                const rn=RANK_NAMES[c.rank], ss=SUITS[c.suit];
                                const isRed=c.suit===1||c.suit===3;
                                if(isRed) el.classList.add('red');
                                el.textContent=rn+ss;
                                if(reveal && gameInProgress) el.classList.add('reveal-anim');
                            } else {
                                el.classList.add('face-down');
                            }
                            mc.appendChild(el);
                        });
                    }
                });

                const me = players[humanPlayerIndex];
                selfChipsEl.textContent = '💰 '+me.chips;
                if(me.cards && me.cards.length===3) {
                    selfCardsRow.innerHTML = '';
                    const ht = getHandType(me.cards);
                    const labelMap = {baoz:'hand-baoz',tonghuashun:'hand-tonghuashun',tonghua:'hand-tonghua',shunzi:'hand-shunzi',duizi:'hand-duizi',danzhang:'hand-danzhang'};
                    selfHandLabel.textContent = ht.name;
                    selfHandLabel.className = 'self-hand-label '+(labelMap[ht.type]||'');
                    me.cards.forEach((c,i)=>{
                        const el=document.createElement('div');
                        el.className='big-card face-up flip-in';
                        el.style.animationDelay = i*0.1+'s';
                        const rn=RANK_NAMES[c.rank], ss=SUITS[c.suit];
                        const isRed=c.suit===1||c.suit===3;
                        const cc=isRed?'red':'black';
                        el.innerHTML=`
                            <span class="card-corner top-left ${cc}">${rn}<br>${ss}</span>
                            <span class="card-rank-big ${cc}">${rn}</span>
                            <span class="card-suit-big ${cc}">${ss}</span>
                            <span class="card-corner bottom-right ${cc}">${rn}<br>${ss}</span>
                        `;
                        selfCardsRow.appendChild(el);
                    });
                }
                selfStatus.textContent = me.status==='folded'?'你已弃牌':me.status==='eliminated'?'你已出局':'';
                potDisplay.textContent = pot;
                currentBetDisplay.textContent = currentBet;
                roundLabel.textContent = '第 '+roundNumber+' 局';
                dealerLabel.textContent = '庄: '+players[dealerIndex].name;
                updateButtons();
            }

            function updateButtons() {
                const me = players[humanPlayerIndex];
                const canAct = gameInProgress && me.status==='active' && getActivePlayers().length>1;
                btnFold.disabled = !canAct;
                btnCall.disabled = !canAct || me.chips < currentBet;
                btnRaise.disabled = !canAct || me.chips < RAISE_AMT;
                btnCompare.disabled = !canAct || getActiveAIs().length===0;
                if(canAct) {
                    btnCall.textContent = '👀 跟注 '+currentBet;
                    btnRaise.textContent = '🔥 加注 '+RAISE_AMT;
                }
                if(gameInProgress) { actionRow.style.display='flex'; newRoundRow.style.display='none'; }
            }

            function showToast(emoji,title,detail,dur=2200) {
                if(toastTimer) clearTimeout(toastTimer);
                toastEmoji.textContent=emoji; toastTitle.textContent=title; toastDetail.textContent=detail;
                toastOverlay.classList.add('show');
                toastTimer = setTimeout(()=>toastOverlay.classList.remove('show'), dur);
            }

            function initPlayers() {
                players = [
                    {name:'👤 我',avatar:'👤',chips:INITIAL_CHIPS,cards:[],status:'active',betThisRound:0,isHuman:true},
                    {name:AI_NAMES[0],avatar:AI_AVATARS[0],chips:INITIAL_CHIPS,cards:[],status:'active',betThisRound:0,isHuman:false},
                    {name:AI_NAMES[1],avatar:AI_AVATARS[1],chips:INITIAL_CHIPS,cards:[],status:'active',betThisRound:0,isHuman:false},
                    {name:AI_NAMES[2],avatar:AI_AVATARS[2],chips:INITIAL_CHIPS,cards:[],status:'active',betThisRound:0,isHuman:false}
                ];
                humanPlayerIndex=0; dealerIndex=0;
            }

            function resetForNewRound() {
                players.forEach(p=>{ if(p.chips>0 && p.status!=='eliminated') p.status='active'; p.cards=[]; p.betThisRound=0; });
                players.forEach(p=>{ if(p.chips<=0) p.status='eliminated'; });
                pot=0; currentBet=BASE_BET;
                if(getActivePlayers().length<2) return false;
                dealerIndex = (dealerIndex+1)%players.length;
                while(players[dealerIndex].status==='eliminated') dealerIndex=(dealerIndex+1)%players.length;
                return true;
            }

            function postBlinds() {
                getActivePlayers().forEach(p=>{
                    const bet = Math.min(BASE_BET, p.chips);
                    p.chips -= bet; p.betThisRound += bet; pot += bet;
                });
                currentBet = BASE_BET;
            }

            function startNewRound() {
                if(!resetForNewRound()) { endGame(); return; }
                gameInProgress=true; dealAll(); postBlinds(); roundNumber++;
                actionRow.style.display='flex'; newRoundRow.style.display='none';
                renderAll(); updateButtons();
                if(players[dealerIndex].isHuman===false && players[dealerIndex].status==='active') {
                    setTimeout(()=>aiAct(dealerIndex), 600);
                }
            }

            function endGame() {
                gameInProgress=false;
                actionRow.style.display='none'; newRoundRow.style.display='flex';
                btnNewRound.textContent='🔄 重新开始'; btnNewRound.onclick=fullReset;
                const me=players[humanPlayerIndex];
                if(me.chips<=0) showToast('💸','你输光了！','点击重新开始',3000);
                else if(getActiveAIs().length===0 && players.filter(p=>p.status==='active').length===1 && me.status==='active')
                    showToast('🏆','你赢光了全场！','太厉害了！',3000);
                renderAll();
            }

            function fullReset() {
                initPlayers(); pot=0; currentBet=BASE_BET; roundNumber=1; gameInProgress=false; dealerIndex=0;
                actionRow.style.display='flex'; newRoundRow.style.display='none';
                btnNewRound.textContent='🔄 再来一局'; btnNewRound.onclick=startNewRound;
                dealAll(); postBlinds(); gameInProgress=true;
                renderAll(); updateButtons();
            }

            function resolveCompare(p1,p2) {
                const result = compareTwo(p1,p2);
                const loser = result.winner===p1 ? p2 : (result.winner===p2 ? p1 : null);
                const winner = result.winner;
                if(loser) loser.status='eliminated';
                if(winner===p1 && loser===p2) return {winner:p1,loser:p2,reason:result.reason};
                if(winner===p2 && loser===p1) return {winner:p2,loser:p1,reason:result.reason};
                return {winner:null,loser:null,reason:result.reason};
            }

            function checkGameEnd() {
                const actives = getActivePlayers();
                if(actives.length <= 1) {
                    if(actives.length===1) {
                        actives[0].chips += pot; pot=0;
                        if(actives[0].isHuman) showToast('🎉','你赢了！','赢得底池',2500);
                        else showToast('😞',actives[0].name+' 赢了','赢得底池',2500);
                    }
                    gameInProgress=false; actionRow.style.display='none'; newRoundRow.style.display='flex';
                    btnNewRound.textContent='🔄 再来一局'; btnNewRound.onclick=startNewRound;
                    renderAll(); return true;
                }
                return false;
            }

            function aiAct(aiIndex) {
                if(!gameInProgress) return;
                const ai = players[aiIndex];
                if(ai.status!=='active') return;
                if(getActivePlayers().length<=1) { checkGameEnd(); return; }
                if(getActiveAIs().length===0 && ai.isHuman===false) return;
                const hand = getHandType(ai.cards);
                const level = hand.level;
                const rand = Math.random();
                let action;
                if(level>=5) action = rand<0.7?'raise':'call';
                else if(level===4) action = rand<0.5?'raise':'call';
                else if(level===3) action = rand<0.35?'raise':(rand<0.8?'call':'fold');
                else if(level===2) action = rand<0.2?'raise':(rand<0.6?'call':'fold');
                else action = rand<0.15?'raise':(rand<0.4?'call':'fold');
                if(rand<0.12 && getActivePlayers().length>=2 && level>=3) action='compare';

                if(action==='fold') {
                    ai.status='folded'; renderAll(); if(checkGameEnd()) return;
                } else if(action==='call') {
                    const bet = Math.min(currentBet, ai.chips);
                    ai.chips-=bet; ai.betThisRound+=bet; pot+=bet;
                    renderAll(); if(checkGameEnd()) return;
                } else if(action==='raise') {
                    const bet = Math.min(RAISE_AMT, ai.chips);
                    ai.chips-=bet; ai.betThisRound+=bet; pot+=bet;
                    currentBet = Math.max(currentBet, bet);
                    renderAll(); if(checkGameEnd()) return;
                } else if(action==='compare') {
                    const targets = getActivePlayers().filter(p=>p!==ai);
                    if(targets.length>0) {
                        const target = targets[Math.floor(Math.random()*targets.length)];
                        const result = resolveCompare(ai, target);
                        if(result.winner && result.loser) { result.winner.chips += pot; pot=0; }
                        renderAll();
                        if(result.winner && result.winner.isHuman) showToast('😮',ai.name+' 向你发起比牌！',result.reason+' - 你赢了！',2500);
                        if(result.loser && result.loser.isHuman) showToast('😢',ai.name+' 向你发起比牌！',result.reason+' - 你出局了',2500);
                        if(checkGameEnd()) return;
                    }
                }
                const nextAIs = getActiveAIs();
                if(nextAIs.length>0 && gameInProgress) {
                    const nextAi = nextAIs[0];
                    setTimeout(()=>aiAct(players.indexOf(nextAi)), 500+Math.random()*600);
                }
                updateButtons();
            }

            btnFold.addEventListener('click', ()=>{
                if(!gameInProgress || players[humanPlayerIndex].status!=='active') return;
                players[humanPlayerIndex].status='folded'; renderAll();
                if(checkGameEnd()) return;
                const ais=getActiveAIs(); if(ais.length>0) setTimeout(()=>aiAct(players.indexOf(ais[0])),500);
            });
            btnCall.addEventListener('click', ()=>{
                if(!gameInProgress || players[humanPlayerIndex].status!=='active') return;
                const me=players[humanPlayerIndex];
                const bet=Math.min(currentBet,me.chips); me.chips-=bet; me.betThisRound+=bet; pot+=bet;
                renderAll(); if(checkGameEnd()) return;
                const ais=getActiveAIs(); if(ais.length>0) setTimeout(()=>aiAct(players.indexOf(ais[0])),500);
            });
            btnRaise.addEventListener('click', ()=>{
                if(!gameInProgress || players[humanPlayerIndex].status!=='active') return;
                const me=players[humanPlayerIndex];
                const bet=Math.min(RAISE_AMT,me.chips); me.chips-=bet; me.betThisRound+=bet; pot+=bet;
                currentBet=Math.max(currentBet,bet); renderAll(); if(checkGameEnd()) return;
                const ais=getActiveAIs(); if(ais.length>0) setTimeout(()=>aiAct(players.indexOf(ais[0])),500);
            });
            btnCompare.addEventListener('click', ()=>{
                if(!gameInProgress || players[humanPlayerIndex].status!=='active') return;
                const me=players[humanPlayerIndex];
                const targets=getActiveAIs(); if(targets.length===0) return;
                const target=targets[0];
                const result=resolveCompare(me,target);
                if(result.winner && result.loser) { result.winner.chips += pot; pot=0; }
                renderAll();
                if(result.winner && result.winner.isHuman) showToast('⚔️','比牌结果',result.reason+' - 你赢了！🎉',2500);
                if(result.loser && result.loser.isHuman) showToast('⚔️','比牌结果',result.reason+' - 你出局了...😢',2500);
                if(result.winner && !result.winner.isHuman && result.loser && result.loser.isHuman) showToast('⚔️','比牌结果',result.reason+' - '+result.winner.name+'赢了',2500);
                if(checkGameEnd()) return;
                const ais=getActiveAIs(); if(ais.length>0) setTimeout(()=>aiAct(players.indexOf(ais[0])),500);
            });
            btnNewRound.addEventListener('click', startNewRound);

            function init() {
                initPlayers(); pot=0; currentBet=BASE_BET; roundNumber=1; gameInProgress=false; dealerIndex=0;
                actionRow.style.display='flex'; newRoundRow.style.display='none';
                btnNewRound.textContent='🔄 再来一局'; btnNewRound.onclick=startNewRound;
                dealAll(); postBlinds(); gameInProgress=true;
                renderAll(); updateButtons();
            }
            init();
        })();
    </script>
